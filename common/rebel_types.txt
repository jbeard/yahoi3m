	ideology = any					# No special preference
	area = nation 					# Stays within the nation
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	unit = partisan_brigade
	
	resilient = no
	reinforcing = yes				
	smart = yes
	auto_convert = yes

	spawn_chance = {
		factor = 10

		}

			factor = 0.1

 	}

	}

	}

	}
}

####################################
nationalist_rebels = {	
	area = nation 				# Stays within provinces of their culture
	defection = none
	ideology = any			# No special preference
	independence = any
	defect_delay = 12
	unit_transfer = yes			# Units switch allegiance after enforcing demands.
	unit = militia_brigade
	
	resilient = no
	reinforcing = yes
	smart = yes
	

	spawn_chance = {
		factor = 10
		modifier = {
			factor = 0.0
			NOT = { controlled_by = owner }
		}
	}
 
 	#This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			revolt_risk = 2
		}
		modifier = {
			factor = 1.5
			revolt_risk = 4
		}
		modifier = {
			factor = 1.5
			revolt_risk = 6
		}
		modifier = {
			factor = 5.0
			is_core = REB
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
 	}
	siege_won_effect = {
	}
	

	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	

	# Country scope
	demands_enforced_effect = {
	}
}
